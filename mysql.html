Basic SQL Commands

SQL Code 
CREATE DATABASE school;
USE school;

CREATE TABLE students(
id INT,
name VARCHAR(50),
marks INT
);
Output
Database created successfully Table "students" created

INSERT Data

SQL Code 
INSERT INTO students
VALUES(1,'John',85);
Output
1 row inserted

SELECT Data

SQL Code 
SELECT * FROM students;
Output
ID Name Marks
1 John 85

Filtering Data

SQL Code 
SELECT * FROM students
WHERE marks > 80;
Output
ID Name Marks
1 John 85

Sorting Data

SQL Code 
SELECT * FROM students
ORDER BY marks DESC;
Output
ID Name Marks
1 John 85

Aggregate Functions

SQL Code 
SELECT AVG(marks)
FROM students;
Output
AVG(marks) ----------- 85

GROUP BY

SQL Code 
SELECT marks,COUNT(*)
FROM students
GROUP BY marks;
Output
Marks Count
85 1

Joins

SQL Code 
SELECT students.name, courses.course
FROM students
INNER JOIN courses
ON students.id = courses.student_id;
Output
Name Course
John Python

Views

SQL Code 
CREATE VIEW top_students AS
SELECT name,marks
FROM students
WHERE marks>80;
Output
View created successfully

Stored Procedure

SQL Code 
CREATE PROCEDURE GetStudents()

SELECT * FROM students;
Output
Procedure created successfully

Triggers

SQL Code 
CREATE TRIGGER before_insert
BEFORE INSERT ON students
FOR EACH ROW
SET NEW.marks = 0;
Output
Trigger created successfully
